// AND Gate

module and_gate(input a, input b, output y);

  assign y = a & b;

endmodule



// OR Gate

module or_gate(input a, input b, output y);

  assign y = a | b;

endmodule



// NOT Gate

module not_gate(input a, output y);

  assign y = ~a;

endmodule



// NAND Gate

module nand_gate(input a, input b, output y);

  assign y = ~(a & b);

endmodule



// NOR Gate

module nor_gate(input a, input b, output y);

  assign y = ~(a | b);

endmodule



// XOR Gate

module xor_gate(input a, input b, output y);

  assign y = a ^ b;

endmodule



// XNOR Gate

module xnor_gate(input a, input b, output y);

  assign y = ~(a ^ b);

endmodule

2. Arithmetic Logic Unit (ALU) (alu.v)

Source Code:

module alu(

input [2:0] a,b,

input [1:0] s,

output reg [2:0] y);



always @(*) 

  begin

    case(s) 

    2'b00: y=a&b;

    2'b01: y=a|b;

    2'b10: y=a^b;

    default: y=3'b000;

endcase 

end



endmodule

3. 4-bit Comparator (comp.v)

Source Code:

module comparator(

input [3:0] a,b,

output reg [3:0] Gt,Eq,Lt);



always @(*) begin



if(a<b) begin

Gt<=0;

Eq<=0;

Lt<=1;

end



else if(a>b) begin

Gt<=1;

Eq<=0;

Lt<=0;

end



else if(a==b) begin

Gt<=0;

Eq<=1;

Lt<=0;

end



end



endmodule

7. 2-to-1 Multiplexer (mux2x1.v)

Source Code:

module mux2x1(

    input  a,      // Input 0

    input  b,      // Input 1

    input  sel,    // Select signal

    output y      // Output

);



    assign y = sel ? b : a;



endmodule
