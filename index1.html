module encoder8to3(input [7:0] in, output reg [2:0] out);

        8'b00001000: out = 3'b011;

2. 1-to-2 Demultiplexer (demux1x2.v)

Source Code:

module demux1x2(input in, input en, output reg y0, output reg y1);
always @(*) begin
    if(en) begin
        y0 = in & ~en;
        y1 = in & en;
    end else begin
        y0 = 0; y1 = 0;
    end
end
endmodule

3. 1-to-4 Demultiplexer (demux1x4.v)

Source Code:

module demux1x4(input in, input [1:0] sel, input en, output reg [3:0] out);
always @(*) begin
    out = 4'b0000;
    if(en) begin
        case(sel)
            2'b00: out[0] = in;
            2'b01: out[1] = in;
            2'b10: out[2] = in;
            2'b11: out[3] = in;
        endcase
    end
end
endmodule

6. 8-to-3 Priority Encoder (priority_encoder.v)

module priority_encoder8to3(input [7:0] in, output reg [2:0] out);
always @(*) begin
    if(in[7]) out = 3'b111;
    else if(in[6]) out = 3'b110;
    else if(in[5]) out = 3'b101;
    else if(in[4]) out = 3'b100;
    else if(in[3]) out = 3'b011;
    else if(in[2]) out = 3'b010;
